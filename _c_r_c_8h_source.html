432template <typename CRCType, crcpp_uint16 CRCWidth>
433inline void CRC::Table<CRCType, CRCWidth>::InitTable()
434{
435 // For masking off the bits for the CRC (in the event that the number of bits in CRCType is larger than CRCWidth)
436 static crcpp_constexpr CRCType BIT_MASK((CRCType(1) << (CRCWidth - CRCType(1))) |
437 ((CRCType(1) << (CRCWidth - CRCType(1))) - CRCType(1)));
438
439 // The conditional expression is used to avoid a -Wshift-count-overflow warning.
440 static crcpp_constexpr CRCType SHIFT((CHAR_BIT >= CRCWidth) ? static_cast<CRCType>(CHAR_BIT - CRCWidth) : 0);
441
442 CRCType crc;
443 unsigned char byte = 0;
444
445 // Loop over each dividend (each possible number storable in an unsigned char)
446 do
447 {
448 crc = CRC::CalculateRemainder<CRCType, CRCWidth>(&byte, sizeof(byte), parameters, CRCType(0));
449
450 // This mask might not be necessary; all unit tests pass with this line commented out,
451 // but that might just be a coincidence based on the CRC parameters used for testing.
452 // In any case, this is harmless to leave in and only adds a single machine instruction per loop iteration.
453 crc &= BIT_MASK;
454
455 if (!parameters.reflectInput && CRCWidth < CHAR_BIT)
456 {
457 // Undo the special operation at the end of the CalculateRemainder()
458 // function for non-reflected CRCs < CHAR_BIT.
459 crc = static_cast<CRCType>(crc << SHIFT);
460 }
461
462 table[byte] = crc;
463 }
464 while (++byte);
465}

563template <typename CRCType, crcpp_uint16 CRCWidth>
564inline CRCType CRC::CalculateBits(const void * data, crcpp_size size, const Parameters<CRCType, CRCWidth> & parameters)
565{
566 CRCType remainder = parameters.initialValue;
567
568 // Calculate the remainder on a whole number of bytes first, then call
569 // a special-case function for the remaining bits.
570 crcpp_size wholeNumberOfBytes = size / CHAR_BIT;
571 if (wholeNumberOfBytes > 0)
572 {
573 remainder = CalculateRemainder(data, wholeNumberOfBytes, parameters, remainder);
574 }
575
576 crcpp_size remainingNumberOfBits = size % CHAR_BIT;
577 if (remainingNumberOfBits != 0)
578 {
579 unsigned char lastByte = *(reinterpret_cast<const unsigned char *>(data) + wholeNumberOfBytes);
580 remainder = CalculateRemainderBits(lastByte, remainingNumberOfBits, parameters, remainder);
581 }
582
583 // No need to mask the remainder here; the mask will be applied in the Finalize() function.
584
585 return Finalize<CRCType, CRCWidth>(remainder, parameters.finalXOR, parameters.reflectInput != parameters.reflectOutput);

704template <typename IntegerType>
705inline IntegerType CRC::Reflect(IntegerType value, crcpp_uint16 numBits)
706{
707 IntegerType reversedValue(0);
708
709 for (crcpp_uint16 i = 0; i < numBits; ++i)
710 {
711 reversedValue = static_cast<IntegerType>((reversedValue << 1) | (value & 1));
712 value = static_cast<IntegerType>(value >> 1);
713 }
714
715 return reversedValue;
716}

727template <typename CRCType, crcpp_uint16 CRCWidth>
728inline CRCType CRC::Finalize(CRCType remainder, CRCType finalXOR, bool reflectOutput)
729{
730 // For masking off the bits for the CRC (in the event that the number of bits in CRCType is larger than CRCWidth)
731 static crcpp_constexpr CRCType BIT_MASK = (CRCType(1) << (CRCWidth - CRCType(1))) |
732 ((CRCType(1) << (CRCWidth - CRCType(1))) - CRCType(1));
733
734 if (reflectOutput)
735 {
736 remainder = Reflect(remainder, CRCWidth);
737 }
738
739 return (remainder ^ finalXOR) & BIT_MASK;
740}

759template <typename CRCType, crcpp_uint16 CRCWidth>

761{
762 // For masking off the bits for the CRC (in the event that the number of bits in CRCType is larger than CRCWidth)

767
768 if (reflectOutput)

790 // This static_assert is put here because this function will always be compiled in no matter what
791 // the template parameters are and whether or not a table lookup or bit-by-bit algorithm is used.
792 static_assert(::std::numeric_limits<CRCType>::digits >= CRCWidth, "CRCType is too small to contain a CRC of width CRCWidth.");
793#else
794 // Catching this compile-time error is very important. Sadly, the compiler error will be very cryptic, but it's
795 // better than nothing.
796 enum { static_assert_failed_CRCType_is_too_small_to_contain_a_CRC_of_width_CRCWidth = 1 / (::std::numeric_limits<CRCType>::digits >= CRCWidth ? 1 : 0) };
797#endif
798
799 const unsigned char * current = reinterpret_cast<const unsigned char *>(data);
800
801 // Slightly different implementations based on the parameters. The current implementations try to eliminate as much
802 // computation from the inner loop (looping over each bit) as possible.
803 if (parameters.reflectInput)
804 {
805 CRCType polynomial = CRC::Reflect(parameters.polynomial, CRCWidth);
806 while (size--)
807 {
808 remainder = static_cast<CRCType>(remainder ^ *current++);
809
810 // An optimizing compiler might choose to unroll this loop.
811 for (crcpp_size i = 0; i < CHAR_BIT; ++i)
812 {
813#ifdef CRCPP_BRANCHLESS
814 // Clever way to avoid a branch at the expense of a multiplication. This code is equivalent to the following:
815 // if (remainder & 1)
816 // remainder = (remainder >> 1) ^ polynomial;
817 // else
818 // remainder >>= 1;
819 remainder = static_cast<CRCType>((remainder >> 1) ^ ((remainder & 1) * polynomial));
820#else
821 remainder = static_cast<CRCType>((remainder & 1) ? ((remainder >> 1) ^ polynomial) : (remainder >> 1));
822#endif
823 }
824 }
825 }
826 else if (CRCWidth >= CHAR_BIT)
827 {
828 static crcpp_constexpr CRCType CRC_WIDTH_MINUS_ONE(CRCWidth - CRCType(1));
829#ifndef CRCPP_BRANCHLESS
830 static crcpp_constexpr CRCType CRC_HIGHEST_BIT_MASK(CRCType(1) << CRC_WIDTH_MINUS_ONE);
831#endif
832 // The conditional expression is used to avoid a -Wshift-count-overflow warning.
833 static crcpp_constexpr CRCType SHIFT((CRCWidth >= CHAR_BIT) ? static_cast<CRCType>(CRCWidth - CHAR_BIT) : 0);
834
835 while (size--)
836 {
837 remainder = static_cast<CRCType>(remainder ^ (static_cast<CRCType>(*current++) << SHIFT));
838
839 // An optimizing compiler might choose to unroll this loop.
840 for (crcpp_size i = 0; i < CHAR_BIT; ++i)
841 {
842#ifdef CRCPP_BRANCHLESS
843 // Clever way to avoid a branch at the expense of a multiplication. This code is equivalent to the following:
844 // if (remainder & CRC_HIGHEST_BIT_MASK)
845 // remainder = (remainder << 1) ^ parameters.polynomial;
846 // else
847 // remainder <<= 1;
848 remainder = static_cast<CRCType>((remainder << 1) ^ (((remainder >> CRC_WIDTH_MINUS_ONE) & 1) * parameters.polynomial));
849#else
850 remainder = static_cast<CRCType>((remainder & CRC_HIGHEST_BIT_MASK) ? ((remainder << 1) ^ parameters.polynomial) : (remainder << 1));
851#endif
852 }
853 }
854 }
855 else
856 {
857 static crcpp_constexpr CRCType CHAR_BIT_MINUS_ONE(CHAR_BIT - 1);
858#ifndef CRCPP_BRANCHLESS
859 static crcpp_constexpr CRCType CHAR_BIT_HIGHEST_BIT_MASK(CRCType(1) << CHAR_BIT_MINUS_ONE);
860#endif
861 // The conditional expression is used to avoid a -Wshift-count-overflow warning.
862 static crcpp_constexpr CRCType SHIFT((CHAR_BIT >= CRCWidth) ? static_cast<CRCType>(CHAR_BIT - CRCWidth) : 0);
863
864 CRCType polynomial = static_cast<CRCType>(parameters.polynomial << SHIFT);
865 remainder = static_cast<CRCType>(remainder << SHIFT);
866
867 while (size--)
868 {
869 remainder = static_cast<CRCType>(remainder ^ *current++);
870
871 // An optimizing compiler might choose to unroll this loop.
872 for (crcpp_size i = 0; i < CHAR_BIT; ++i)
873 {
874#ifdef CRCPP_BRANCHLESS
875 // Clever way to avoid a branch at the expense of a multiplication. This code is equivalent to the following:
876 // if (remainder & CHAR_BIT_HIGHEST_BIT_MASK)
877 // remainder = (remainder << 1) ^ polynomial;
878 // else
879 // remainder <<= 1;
880 remainder = static_cast<CRCType>((remainder << 1) ^ (((remainder >> CHAR_BIT_MINUS_ONE) & 1) * polynomial));
881#else
882 remainder = static_cast<CRCType>((remainder & CHAR_BIT_HIGHEST_BIT_MASK) ? ((remainder << 1) ^ polynomial) : (remainder << 1));
883#endif
884 }
885 }
886
887 remainder = static_cast<CRCType>(remainder >> SHIFT);
888 }
889
890 return remainder;
891}

903template <typename CRCType, crcpp_uint16 CRCWidth>
904inline CRCType CRC::CalculateRemainder(const void * data, crcpp_size size, const Table<CRCType, CRCWidth> & lookupTable, CRCType remainder)
905{
906 const unsigned char * current = reinterpret_cast<const unsigned char *>(data);
907
908 if (lookupTable.GetParameters().reflectInput)
909 {
910 while (size--)
911 {
912#if defined(_MSC_VER) && !defined(__clang__) && !defined(__INTEL_COMPILER)
913 // Disable warning about data loss when doing (remainder >> CHAR_BIT) when
914 // remainder is one byte long. The algorithm is still correct in this case,
915 // though it's possible that one additional machine instruction will be executed.
916# pragma warning (push)
917# pragma warning (disable : 4333)
918#endif
919 remainder = static_cast<CRCType>((remainder >> CHAR_BIT) ^ lookupTable[static_cast<unsigned char>(remainder ^ *current++)]);
920#if defined(_MSC_VER) && !defined(__clang__) && !defined(__INTEL_COMPILER)
921# pragma warning (pop)
922#endif
923 }
924 }
925 else if (CRCWidth >= CHAR_BIT)
926 {
927 // The conditional expression is used to avoid a -Wshift-count-overflow warning.
928 static crcpp_constexpr CRCType SHIFT((CRCWidth >= CHAR_BIT) ? static_cast<CRCType>(CRCWidth - CHAR_BIT) : 0);
929
930 while (size--)
931 {
932 remainder = static_cast<CRCType>((remainder << CHAR_BIT) ^ lookupTable[static_cast<unsigned char>((remainder >> SHIFT) ^ *current++)]);
933 }
934 }
935 else
936 {
937 // The conditional expression is used to avoid a -Wshift-count-overflow warning.
938 static crcpp_constexpr CRCType SHIFT((CHAR_BIT >= CRCWidth) ? static_cast<CRCType>(CHAR_BIT - CRCWidth) : 0);
939
940 remainder = static_cast<CRCType>(remainder << SHIFT);
941
942 while (size--)
943 {
944 // Note: no need to mask here since remainder is guaranteed to fit in a single byte.
945 remainder = lookupTable[static_cast<unsigned char>(remainder ^ *current++)];
946 }
947
948 remainder = static_cast<CRCType>(remainder >> SHIFT);
949 }
950
951 return remainder;
952}
